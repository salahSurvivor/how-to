const os = require('os');

//platform
console.log(os.platform());

//cpu Arch 
console.log(os.arch());

//cpu core info 
console.log(os.cpus());

// free memory
console.log(os.freemem());

// total memory
console.log(os.totalmem());

//home directory
console.log(os.homedir());

//Uptime
console.log(os.uptime());
